145  KDNodePtr build_tree_recursive(std::vector< Point<d, T> >& pointvec, int l, int r, int id)
146  {
147  if(r<=l) return nullptr;
148  int m = l+(r-l)/2;
149  // O(n) partition such that elements to the left are < than element at pivot position and elements to the right are >= to element at pivot position
150  std::nth_element(pointvec.begin()+l, pointvec.begin()+m, pointvec.begin()+r, [&id](Point<d, T>& a, Point<d, T>& b){
